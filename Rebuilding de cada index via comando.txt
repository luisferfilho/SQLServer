SELECT 'ALTER INDEX ' + QUOTENAME(name) + ' ON ' + QUOTENAME(OBJECT_SCHEMA_NAME(object_id)) + '.' + QUOTENAME(OBJECT_NAME(object_id)) + ' REBUILD PARTITION = ALL WITH (FILLFACTOR = 90)' FROM sys.indexes WHERE OBJECTPROPERTY(object_id, 'IsMSShipped') = 0 AND OBJECTPROPERTY(object_id, 'IsTable') = 1 ORDER BY CASE type_desc WHEN 'CLUSTERED' THEN 1 ELSE 2 END
